## 5a - Task 1:
SELECT orders.order_id, SUM(price * quantity) AS total_price FROM orders FULL OUTER JOIN line_items ON orders.order_id = line_items.order_id FULL OUTER JOIN products ON line_items.product_id = products.product_id GROUP BY orders.order_id ORDER BY orders.order_id ASC LIMIT 5;

## 5a - Task 2:
SELECT c.customer_name, AVG(t.order_total) AS average_order_price FROM customers AS c 
JOIN (SELECT o.customer_id, SUM(p.price * li.quantity) AS order_total FROM orders AS o JOIN line_items AS li ON o.order_id = li.order_id JOIN products AS p ON li.product_id = p.product_id GROUP BY o.customer_id, o.order_id) AS t ON c.customer_id = t.customer_id GROUP BY c.customer_id, c.customer_name ORDER BY c.customer_name;
